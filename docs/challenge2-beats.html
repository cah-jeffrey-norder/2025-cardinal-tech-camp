Beats




  // Four on the Floor
  $: sound("[bd bd bd bd]").bank('RolandTR909')

  // Tildes are RESTS ~
  $: sound("[~ sd ~ [~ sd]]").bank('RolandTR909')

  // Multipliers (*) can make things go faster
  $: sound("[hh hh hh hh]*2").bank('RolandTR909')

  $: sound("bd hh sd [ht mt lt]")

Special Characters
    [] - play all notes in a single beat
    <> - Each Cycle
    ~  - rest
    *  - speed up
    /  - slow down
    !  - replicate
    @  - elongate
    ,  - parallel
    $  - play in parallel threads
    _  - mute a line
    SPACE - sequence

DRUM LIST
    bd = bass drum
    hh = high hat
    oh = open hat
    sd = snare drum
    rd = ride
    cr = crash
    lt = low tom
    mt = med tom
    ht = high tom
    cp = clap
    sh = shaker
    cb = cowbell
    tb = tambourine
